\begin{frame}{Overview}
  \begin{itemize}
    \item State the narrative in 3--5 sections, not a wall of subsections.
    \item Use overview or progress frames sparingly and keep them visually light.
    \item Highlight the current section by weight, not by introducing new colors.
  \end{itemize}
\end{frame}

\section{Situation}

\begin{frame}{Problem Framing}
  \begin{block}{Why this matters}
    Lead with one sentence that establishes the practical or scientific importance.
  \end{block}

  \vspace{0.4cm}
  \begin{itemize}
    \item Use 3 short bullets that read as claims.
    \item Keep the left margin clean and preserve large whitespace.
    \item Cite external quantitative claims in a small source line near the bottom.
  \end{itemize}
\end{frame}

\begin{frame}{Two-Column Comparison}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.49\textwidth}
      \begin{block}{Current State}
        \begin{itemize}
          \item Summarize the baseline or conventional approach.
          \item Add one limitation per bullet.
          \item Avoid paragraph-sized prose inside the panel.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.49\textwidth}
      \begin{highlightbox}{What changes}
        \begin{itemize}
          \item Use the right column for the improved view or takeaway.
          \item Keep the language claim-like and scan-friendly.
          \item Use gold only for selected emphasis, not the whole deck structure.
        \end{itemize}
      \end{highlightbox}
    \end{column}
  \end{columns}
\end{frame}

\section{Closing}

\begin{frame}{Closing}
  \begin{block}{What to remember}
    Summarize the final decision or result in one sentence.
  \end{block}
  \vspace{0.5cm}
  \begin{itemize}
    \item Main conclusion
    \item One risk or limitation
    \item Next step or discussion point
  \end{itemize}
\end{frame}
